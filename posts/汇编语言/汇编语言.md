# 技术控制循环
## 不使用LOOP
```assembly {9-11}
xor esi,esi  ;初始化ESI为0，用于数组索引，也可以用mov esi,0
mov eax,array[esi]  ;eax保存最大值，先将数组的第一个元素存入EAX
again:
add esi,1  ;增量计数（ESI+1,指向下一个数组元素）
add esi,array[esi*4]   ;比较EAX和当前数组元素，由于数组元素是4个字节，所以esi*4寻址
jge next    ;如果eax≥当前元素，则跳转到next
mov eax,array[esi*4]    ;否则更新EAX为当前较大的元素
next:
cmp esi,count-1           ;比较索引和“元素个数－1”（判断是否遍历完所有元素）
jb again                  ;如果索引小于count-1,继续循环
mov max,eax               ;保存最大值（将最终的最大值存入max变量）
```
流程如果是<br>
xx-YES－>xxx－>xxxx<br>
xx------NO---->xxxx
时，可以写成
```assembly
xx
cmp
jmp a   ;NO则跳
xxx
a:
xxxx
```
绝大多数的循环都是判断后循环(先行判断)
```assembly "跳出" "循环" "满足" "不满足"
先行判断
again:
    cmp string[edx],0   ;判断跳出，如果跳出条件满足就跳出    
    jz done   
    inc ebx         
    jmp agiain            
done:
    exit
先行循环
again:
    inc ebx
    cmp string[edx],0       ;判断循环，如果循环条件不满足就继续执行（不跳转循环）
    jnz again        
done:     ;这个done标签此时是多余的
    exit
```
## 使用LOOP指令
```assembly title="使用LOOP减量循环"
mov ecx,num
label:
    loop_code
looop lable
next code
```
loop指令使用ecx减量循环
:::caution
ecx=0时循环2^32，因为LOOP指令是先行循环
<br>可改用jecxz<br>
与LOOP相似，但是只有在ecx!=0时才生效
:::

### 寄存器传递参数
子程序
​```assembly
htoasc  proc
    add al,0fh
    or al,30h   ;由于al的高位都是0，相或等价于相加（与0或得到其本身）
    cmp al,39h
    jbe htoend
    add al,7
htoend: ret
htoasc endp
```
还有一种写法,使用映射表
```
htoasc proc
    and eax,0fh
    mov al,ASCII[eax]
    ret
ASCII byte '0123456789ABCDEF'
htoend: ret
htoasc endp
```
